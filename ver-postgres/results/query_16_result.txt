SELECT c.customer_code, c.full_name
FROM customers c
WHERE c.customer_id IN (
    SELECT p.customer_id
    FROM policies p
    WHERE p.policy_id IN (
        SELECT cl.policy_id
        FROM claims cl
        WHERE cl.paid_amount > 3000000
    )
)
ORDER BY c.customer_code;
